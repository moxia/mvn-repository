##使用：##
```xml
<repositories>
    <repository>
        <id>mvn-repository</id>
        <url>http://git.oschina.net/xoxoxoxoxoxoxoxo/mvn-repository/tree/master/releases</url>
    </repository>
</repositories>

<!-- 如使用jpinyin -->
<dependencies>
    <dependency>
        <groupId>opensource</groupId>
        <artifactId>jpinyin</artifactId>
        <version>1.0</version>
    </dependency>
    <!-- ... -->
</dependencies>
```
